2138. 将字符串拆分为若干长度为 k 的组#

难度：简单

补全分割

class Solution {
    public String[] divideString(String s, int k, char fill) {
    	int num = s.length() % k;
    	if(num != 0) {
    		num = k - num;
    		while(num-- > 0) {
    			s += fill;
    		}
    	}
    	int len = s.length() / k;
    	String[] ans = new String[len];
    	for(int i = 0; i < len; i++) {
    		ans[i] = s.substring(i*k, (i + 1) * k);
    	}
    	return ans;
    }
}

2139. 得到目标值的最少行动次数#

难度：中等

贪心

class Solution {
    public int minMoves(int target, int maxDoubles) {
    	int cnt = 0;
    	while(target > 1) {
    		if(maxDoubles > 0) {
    			if((target & 1) == 1) target--;
    			else {
    				target >>= 1;
    				maxDoubles--;
    			}
    			cnt++;
    		}else {
    			return target + cnt - 1;
    		}
    	}
    	return cnt;
    }
}

2140. 解决智力问题#

难度：中等

逆序dp

class Solution {
    public long mostPoints(int[][] questions) {
    	int n = questions.length;
    	long[] dp = new long[n + 1];
    	for(int i = n -1; i >= 0; i--) {
    		if(questions[i][1] + i + 1 < n) {
    			dp[i] = Math.max(dp[i + 1], dp[questions[i][1] + i + 1] + questions[i][0]);
    		} else {
    			dp[i] = Math.max(dp[i + 1], questions[i][0]);
    		}
    	}
    	return dp[0];
    }
}

2133. 检查是否每一行每一列都包含全部整数#

难度：简单

HashSet

class Solution {
    public boolean checkValid(int[][] matrix) {
    	int len = matrix.length;
    	HashSet<Integer> h1 = new HashSet<>();
    	HashSet<Integer> h2 = new HashSet<>();
    	for(int i = 0; i < len; i++) {
    		for(int j = 0; j < len; j++) {
    			h1.add(matrix[i][j]);
    			h2.add(matrix[j][i]);
    		}
    		if(h1.size() != len || h2.size() != len) return false;
    		h1.clear();
    		h2.clear();
    	}
    	return true;
    }
}

2129. 将标题首字母大写#

难度：简单

Api+Ascii

class Solution {
    public String capitalizeTitle(String title) {
    	String[] str = title.split(" ");
    	for(int i = 0; i < str.length; i++) {
    		if(str[i].length() < 3) {
    			str[i] = str[i].toLowerCase();
    		} else {
    			char[] c = str[i].toLowerCase().toCharArray();
    			c[0] &= ~32;
    			str[i] = String.valueOf(c);
    		}
    	}
    	return String.join(" ", str);
    }
}

2130. 链表最大孪生和#

难度：中等

链表转数组+双指针

class Solution {
    public int pairSum(ListNode head) {
    	if(head.next == null) return 0;
    	List<Integer> list = new ArrayList<>();
    	while(head != null) {
    		list.add(head.val);
    		head = head.next;
    	}
    	Integer[] num = list.toArray(new Integer[0]);
    	int mid = num.length >> 1;
    	int max = 0;
    	for(int i = mid - 1; i >= 0; i--){
    		max = Math.max(max, num[i] + num[mid++]);
    	}
    	return max;
    }
}

2124. 检查是否所有 A 都在 B 之前#

难度：简单

标记

class Solution {
    public boolean checkString(String s) {
    	char[] cs = s.toCharArray();
    	boolean na = false;
    	for(char c : cs) {
    		if(c == 'a' && na) return false;
    		if(c != 'a') na = true;
    	}
    	return true;
    }
}

2125. 银行中的激光束数量#

难度：中等

cnt * pre

class Solution {
    public int numberOfBeams(String[] bank) {
     	int ans = 0;
    	int pre = 0;
    	for(String s : bank) {
    		int cnt = 0;
    		char[] cs = s.toCharArray();
    		for(int i = 0; i < cs.length; i++) {
    			cnt += cs[i] == '1' ? 1 : 0;
    		}
    		if(cnt != 0) {
    			ans += cnt * pre;
    			pre = cnt;
    		}
    	}
    	return ans;
    }
}

2126. 摧毁小行星#

难度：中等

排序+贪心

class Solution {
    public boolean asteroidsDestroyed(int mass, int[] asteroids) {
    	long mas = mass;
    	Arrays.sort(asteroids);
    	for(int i : asteroids) {
    		if(mas < i) return false;
    		mas += i;
    	}
    	return true;
    }
}

class Solution {
    public boolean isSameAfterReversals(int num) {
        return !(num != 0 && num % 10 == 0);

}

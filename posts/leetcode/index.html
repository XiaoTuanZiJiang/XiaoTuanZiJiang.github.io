### [2.14-540. 有序数组中的单一元素](https://leetcode-cn.com/problems/single-element-in-a-sorted-array/)
难度：<font color=#FFA119>中等</font>

>二分

	class Solution {
	    public int singleNonDuplicate(int[] nums) {
	        int n = nums.length;
	        int l = 0,r = n - 1;
	        while(l < r){
	            int mid = (l + r) >> 1;
	            if((mid & 1) == 0){
	                if (mid + 1 < n && nums[mid] == nums[mid + 1]) l = mid + 1;
	                else r = mid;
	            }else{
	                if (mid - 1 >= 0 && nums[mid - 1] == nums[mid]) l = mid + 1;
	                else r = mid;
	            }
	        }
	        return nums[r];
	    }
	}
### [2.11-1984. 学生分数的最小差值](https://leetcode-cn.com/problems/minimum-difference-between-highest-and-lowest-of-k-scores/)
难度：<font color=#5AB726 >简单</font>

>排序+动态窗口

	class Solution {
	    public int minimumDifference(int[] nums, int k) {
	        Arrays.sort(nums);
	        int ans = Integer.MAX_VALUE;
	        for (int i = 0; i <= nums.length - k; i++) {
	            ans = Math.min(ans, nums[i + k - 1] - nums[i]);
	        }
	        return ans;
	    }
	}

	        for (int[] r : rectangles) {

### [1.30-884. 两句话中的不常见单词](https://leetcode-cn.com/problems/uncommon-words-from-two-sentences/)
难度：<font color=#5AB726 >简单</font>

>HashMap

	class Solution {
	    public String[] uncommonFromSentences(String s1, String s2) {
	        Map<String, Integer> map = new HashMap<>();
	        count(s1, map);
	        count(s2, map);
	        List<String> list = new ArrayList<>();
	        map.forEach((k, v) -> {
	            if (v == 1) {
	                list.add(k);
	            }
	        });
	        return list.toArray(new String[list.size()]);
	    }
	
	    private void count(String s, Map<String, Integer> map) {
	        for (String a : s.split(" ")) {
	            map.put(a, map.getOrDefault(a, 0) + 1);
	        }
	    }
	}

	class Solution {

	        Arrays.sort(properties,(a,b)->{
	            return a[0] == b[0] ? (a[1] - b[1]) : (b[0] - a[0]);
	        });
	        int cnt = 0;
	        int temp = 0;
	        for (int[] p : properties) {
	            if(p[1] < temp) cnt++;
	            else temp = p[1];
	        }
	        return cnt;
	    }
	}
### [1.27-2047. 句子中的有效单词数](https://leetcode-cn.com/problems/number-of-valid-words-in-a-sentence/)
难度：<font color=#5AB726 >简单</font>

>模拟

	class Solution {
	    public int countValidWords(String sentence) {
	        String[] ss = sentence.split(" ");
	        int ans = 0;
	        for (String s : ss) if (check(s)) ans++;
	        return ans;
	    }
	    boolean check(String s) {
	        int n = s.length();
	        if (n == 0) return false;
	        for (int i = 0, c1 = 0, c2 = 0; i < n; i++) {
	            char c = s.charAt(i);
	            if (Character.isDigit(c)) return false;
	            if (c == ' ') return false;
	            if (c == '-' && ++c1 >= 0) {
	                if (c1 > 1 || (i == 0 || i == n - 1)) return false;
	                if (!Character.isLetter(s.charAt(i - 1)) || !Character.isLetter(s.charAt(i + 1))) return false;
	            }
	            if ((c == '!' || c == '.' || c == ',') && ++c2 >= 0) {
	                if (c2 > 1 || (i != n - 1)) return false;
	            }
	        }
	        return true;
	    }
	}

### [1.19-219.存在重复元素II](https://leetcode-cn.com/problems/contains-duplicate-ii)
难度：<font color=#5AB726 >简单</font>

>HashMap

    class Solution {
        public boolean containsNearbyDuplicate(int[] nums, int k) {
            HashMap<Integer, Integer> hashMap = new HashMap<>();
            for(int i = 0; i < nums.length; i++) {
                int num = nums[i];
                if(hashMap.containsKey(num)) {
                    return true;
                }
                hashMap.put(num, null);
                if(hashMap.size() > k) {
                    hashMap.remove(nums[i - k]);
                }
            }
            return false;
        }
    }

### [2138. 将字符串拆分为若干长度为 k 的组](https://leetcode-cn.com/problems/divide-a-string-into-groups-of-size-k/)  
难度：<font color=#5AB726 >简单</font>
>补全分割

	class Solution {
	    public String[] divideString(String s, int k, char fill) {
	    	int num = s.length() % k;
	    	if(num != 0) {
	    		num = k - num;
	    		while(num-- > 0) {
	    			s += fill;
	    		}
	    	}
	    	int len = s.length() / k;
	    	String[] ans = new String[len];
	    	for(int i = 0; i < len; i++) {
	    		ans[i] = s.substring(i*k, (i + 1) * k);
	    	}
	    	return ans;
	    }
	}
	
### [2139. 得到目标值的最少行动次数](https://leetcode-cn.com/problems/divide-a-string-into-groups-of-size-k/)  
难度：<font color=#FFA119>中等</font>
>贪心

	class Solution {
	    public int minMoves(int target, int maxDoubles) {
	    	int cnt = 0;
	    	while(target > 1) {
	    		if(maxDoubles > 0) {
	    			if((target & 1) == 1) target--;
	    			else {
	    				target >>= 1;
	    				maxDoubles--;
	    			}
	    			cnt++;
	    		}else {
	    			return target + cnt - 1;
	    		}
	    	}
	    	return cnt;
	    }
	}
	
### [2140. 解决智力问题](https://leetcode-cn.com/problems/solving-questions-with-brainpower/)  
难度：<font color=#FFA119>中等</font>
>逆序dp

	class Solution {
	    public long mostPoints(int[][] questions) {
	    	int n = questions.length;
	    	long[] dp = new long[n + 1];
	    	for(int i = n -1; i >= 0; i--) {
	    		if(questions[i][1] + i + 1 < n) {
	    			dp[i] = Math.max(dp[i + 1], dp[questions[i][1] + i + 1] + questions[i][0]);
	    		} else {
	    			dp[i] = Math.max(dp[i + 1], questions[i][0]);
	    		}
	    	}
	    	return dp[0];
	    }
	}

### [2133. 检查是否每一行每一列都包含全部整数](https://leetcode-cn.com/problems/check-if-every-row-and-column-contains-all-numbers/)  
难度：<font color=#5AB726 >简单</font>
>HashSet

	class Solution {
	    public boolean checkValid(int[][] matrix) {
	    	int len = matrix.length;
	    	HashSet<Integer> h1 = new HashSet<>();
	    	HashSet<Integer> h2 = new HashSet<>();
	    	for(int i = 0; i < len; i++) {
	    		for(int j = 0; j < len; j++) {
	    			h1.add(matrix[i][j]);
	    			h2.add(matrix[j][i]);
	    		}
	    		if(h1.size() != len || h2.size() != len) return false;
	    		h1.clear();
	    		h2.clear();
	    	}
	    	return true;
	    }
	}

难度：<font color=#5AB726 >简单</font>
>Api+Ascii

	class Solution  {
	    public String capitalizeTitle(String title) {
	    	String[] str = title.split(" ");
	    	for(int i = 0; i < str.length; i++) {
	    		if(str[i].length() < 3) {
	    			str[i] = str[i].toLowerCase();
	    		} else {
	    			char[] c = str[i].toLowerCase().toCharArray();
	    			c[0] &= ~32;
	    			str[i] = String.valueOf(c);
	    		}
	    	}
	    	return String.join(" ", str);
	    }
	}
### [2130. 链表最大孪生和](https://leetcode-cn.com/problems/maximum-twin-sum-of-a-linked-list/)  
难度：<font color=#FFA119>中等</font>
>链表转数组+双指针

	class Solution   {
	    public int pairSum(ListNode head) {
	    	if(head.next == null) return 0;
	    	List<Integer> list = new ArrayList<>();
	    	while(head != null) {
	    		list.add(head.val);
	    		head = head.next;
	    	}
	    	Integer[] num = list.toArray(new Integer[0]);
	    	int mid = num.length >> 1;
	    	int max = 0;
	    	for(int i = mid - 1; i >= 0; i--){
	    		max = Math.max(max, num[i] + num[mid++]);
	    	}
	    	return max;
	    }
	}

### [2125. 银行中的激光束数量](https://leetcode-cn.com/problems/number-of-laser-beams-in-a-bank/)  
难度：<font color=#FFA119>中等</font>
>cnt * pre

	class Solution {
	    public int numberOfBeams(String[] bank) {
	     	int ans = 0;
	    	int pre = 0;
	    	for(String s : bank) {
	    		int cnt = 0;
	    		char[] cs = s.toCharArray();
	    		for(int i = 0; i < cs.length; i++) {
	    			cnt += cs[i] == '1' ? 1 : 0;
	    		}
	    		if(cnt != 0) {
	    			ans += cnt * pre;
	    			pre = cnt;
	    		}
	    	}
	    	return ans;
	    }
	}

	class Solution {
	    public int majorityElement(int[] nums) {
	    	int cnt = 0;
	    	int temp = 0;
	    	for(int i : nums) {
	    		if(cnt == 0) {
	    			temp = i;
	    			cnt = 1;
	    		}else {
	    			cnt += temp != i ? -1 : 1;
	    		}
	    	}
	    	return temp;
	    }
	}
